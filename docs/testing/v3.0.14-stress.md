## 1. 单元逻辑(纯函数)

```python
# Python REPL / pytest
import sqlite3
from sheets import dedupe_assign_sheet_tabs

# 模拟 accounts 表
conn = sqlite3.connect(":memory:")
conn.row_factory = sqlite3.Row
conn.execute("""
    CREATE TABLE accounts (
        id INTEGER PRIMARY KEY,
        phone TEXT,
        name TEXT,
        sheet_tab TEXT
    )
""")

# 场景 1: 两个同名「张三」,phone 不同
conn.executescript("""
    INSERT INTO accounts VALUES (1, '+85500000045', '张三', NULL);
    INSERT INTO accounts VALUES (2, '+85500000084', '张三', NULL);
""")
fixed = dedupe_assign_sheet_tabs(conn)
assert fixed == 1, f"应 fix 1 个,实际 {fixed}"
rows = list(conn.execute("SELECT phone, name, sheet_tab FROM accounts ORDER BY phone"))
# phone 字典序:+85586... < +85596...,所以 +85586... 是「第一个」保留 name
assert rows[0]['sheet_tab'] is None, f"第一个不应改,实际 {rows[0]['sheet_tab']}"
assert rows[1]['sheet_tab'] == '张三-6045', f"第二个应是张三-6045,实际 {rows[1]['sheet_tab']}"
print("✓ 场景 1 通过")

# 场景 2: 幂等(再调一次结果不变)
fixed2 = dedupe_assign_sheet_tabs(conn)
assert fixed2 == 0, f"已 fix 应不再 fix,实际 {fixed2}"
print("✓ 场景 2 通过(幂等)")

# 场景 3: 三个同名
conn.execute("INSERT INTO accounts VALUES (3, '+85577001122', '老王', NULL)")
conn.execute("INSERT INTO accounts VALUES (4, '+85577003344', '老王', NULL)")
conn.execute("INSERT INTO accounts VALUES (5, '+85577005566', '老王', NULL)")
fixed3 = dedupe_assign_sheet_tabs(conn)
assert fixed3 == 2, f"3 个老王应 fix 2 个,实际 {fixed3}"
rows = list(conn.execute("SELECT phone, sheet_tab FROM accounts WHERE name='老王' ORDER BY phone"))
assert rows[0]['sheet_tab'] is None, "第一个老王 sheet_tab 应为空"
assert rows[1]['sheet_tab'] == '老王-3344', "第二个老王应 sheet_tab=老王-3344"
assert rows[2]['sheet_tab'] == '老王-5566', "第三个老王应 sheet_tab=老王-5566"
print("✓ 场景 3 通过(三个同名)")

# 场景 4: 已有 sheet_tab 的不动
conn.execute("INSERT INTO accounts VALUES (6, '+85578001111', '小明', '小明-自定义')")
conn.execute("INSERT INTO accounts VALUES (7, '+85578002222', '小明', NULL)")
fixed4 = dedupe_assign_sheet_tabs(conn)
# 7 是 phone 字典序第二个,被 fix
assert fixed4 == 1
rows = list(conn.execute("SELECT phone, sheet_tab FROM accounts WHERE name='小明' ORDER BY phone"))
assert rows[0]['sheet_tab'] == '小明-自定义', "已设的不应被覆盖"
assert rows[1]['sheet_tab'] == '小明-2222', "新的应加后缀"
print("✓ 场景 4 通过(已设 sheet_tab 的不动)")

# 场景 5: 同名只 1 个不动
conn.execute("INSERT INTO accounts VALUES (8, '+85579999999', 'Caer | US', NULL)")
fixed5 = dedupe_assign_sheet_tabs(conn)
assert fixed5 == 0, "单独一个不应 fix"
print("✓ 场景 5 通过(单一 name 不动)")

# 场景 6: name 为空不动
conn.execute("INSERT INTO accounts VALUES (9, '+85580000000', '', NULL)")
conn.execute("INSERT INTO accounts VALUES (10, '+85580000001', '', NULL)")
conn.execute("INSERT INTO accounts VALUES (11, '+85580000002', NULL, NULL)")
fixed6 = dedupe_assign_sheet_tabs(conn)
assert fixed6 == 0, "name 为空不应 fix"
print("✓ 场景 6 通过(空 name 不动)")
```

**验收**: 全部 6 场景 ✓ pass。

## 5. 跨容器并发 race(理论场景)

### 模拟
```bash
# tg-web 和 tg-monitor 同时调 dedupe(并发竞争 UPDATE 同一 id)
for i in {1..10}; do
  docker exec tg-web-dev python3 -c "
import database as db
from sheets import dedupe_assign_sheet_tabs
print(dedupe_assign_sheet_tabs(db.get_conn()))
" &
  docker exec tg-monitor-dev python3 -c "
import database as db
from sheets import dedupe_assign_sheet_tabs
print(dedupe_assign_sheet_tabs(db.get_conn()))
" &
done
wait

# 验证最终 DB 状态一致
docker exec tg-monitor-dev sqlite3 /app/data/monitor.db \
  "SELECT phone, name, sheet_tab FROM accounts WHERE name='测试号' ORDER BY phone;"
```

**验收**:
- 无 SQLite lock error(WAL 模式 + UPDATE WHERE id 单行短锁)
- 最终 DB 状态确定性(按 phone 字典序排,第一个空 sheet_tab,后续都是 `测试号-后4位`),并发 N 次结果跟单次一致
